export const permissions = `import { createHelm, fs, git, grep } from "@bgub/helm";

const agent = createHelm({
  permissions: {
    "fs.readFile": "allow",  // runs immediately
    "fs.*": "ask",            // prompts via callback
    "git.*": "allow",         // all git reads allowed
    "shell.*": "deny",        // no shell access
  },
  onPermissionRequest: async (operation, args) => {
    console.log(\`\${operation} requested with\`, args);
    return true; // or false to deny
  },
}).use(fs()).use(git()).use(grep());

const { content } = await agent.fs.readFile("./package.json");
const { staged, branch } = await agent.git.status();
const { matches } = await agent.grep.search("TODO", { glob: "*.ts" });`;

export const skill = `const weather = defineSkill({
  name: "weather",
  description: "Weather forecast operations",
  operations: {
    forecast: {
      description: "Get the forecast for a city",
      signature: "(city: string) => Promise<{ temp: number; sky: string }>",
      defaultPermission: "allow",
      tags: ["weather", "read"],
      handler: async (city: string): Promise<{ temp: number; sky: string }> => {
        return fetchWeather(city);
      },
    },
  },
});

const agent = createHelm().use(fs()).use(weather);
const { temp, sky } = await agent.weather.forecast("Seattle");`;
